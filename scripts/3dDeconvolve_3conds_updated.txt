## Used this to create emotional-neutral contrast that was not included in afniprocpy

#!/usr/bin/env bash
ROOT="/Volumes/ELEMENTS/imagery"
PARDIR="$ROOT/participants"


for subdir in "$PARDIR"/par_*; do
  SUBJ="${subdir##*/}"
  fmridir="$subdir/fmri"
  
  mkdir -p "$fmridir/PROC_cspline_new"

  indir="$fmridir/PROC_cspline"
  outdir="$fmridir/PROC_cspline_new"

  [[ -d "$indir" ]] || { echo "No PROC dir for $SUBJ, skipping"; continue; }
  cd "$indir" || continue

  echo "=== Processing $SUBJ ==="

  # Detect O1 vs O2 prefix for this subject
  prefix=""
  for p in O1 O2; do
    if [[ -f "$fmridir/${p}.pleasant.LOCAL.txt" ]]; then
      prefix="$p"
      break
    fi
  done

  if [[ -z "$prefix" ]]; then
    echo "** No timing files found for $SUBJ in $indir, skipping"
    continue
  fi

  echo "   Using timing prefix: $prefix"


  # Dynamically build -ortvec args based on which mot_demean files exist
  ortvec_args=()
  for r in 01 02; do
    f="mot_demean.r${r}.1D"
    if [[ -f "$f" ]]; then
      ortvec_args+=( -ortvec "$f" "mot_demean_r${r}" )
    fi
  done

  if [[ ${#ortvec_args[@]} -eq 0 ]]; then
    echo "** No mot_demean files found for $SUBJ, skipping"
    continue
  fi


  3dDeconvolve \
    -input pb04.${SUBJ}.r*.scale+tlrc.HEAD \
    -mask "$PARDIR/par_01/fmri/PROC_cspline/mask_group+tlrc" \
    -censor "$indir/censor_${SUBJ}_combined_2.1D" \
    "${ortvec_args[@]}" \
    -polort A \
    -num_stimts 3 \
    -stim_times 1 "$fmridir/${prefix}.pleasant.LOCAL.txt"   'CSPLINzero(0,21,8)' \
    -stim_label 1 pleasant \
    -stim_times 2 "$fmridir/${prefix}.neutral.LOCAL.txt"    'CSPLINzero(0,21,8)' \
    -stim_label 2 neutral \
    -stim_times 3 "$fmridir/${prefix}.unpleasant.LOCAL.txt" 'CSPLINzero(0,21,8)' \
    -stim_label 3 unpleasant \
    -iresp 1 "$outdir/iresp_pleasant.${SUBJ}" \
    -iresp 2 "$outdir/iresp_neutral.${SUBJ}" \
    -iresp 3 "$outdir/iresp_unpleasant.${SUBJ}" \
    -fout -tout \
    -gltsym 'SYM: +pleasant -neutral'               -glt_label 1 pleasant-neutral \
    -gltsym 'SYM: +unpleasant -neutral'             -glt_label 2 unpleasant-neutral \
    -gltsym 'SYM: +pleasant -unpleasant'            -glt_label 3 pleasant-unpleasant \
    -gltsym 'SYM: +pleasant +unpleasant -2*neutral' -glt_label 4 emotional-neutral \
      -x1D "$outdir/X.xmat.1D" \
    -xjpeg "$outdir/X.jpg" \
    -x1D_uncensored "$outdir/X.nocensor.xmat.1D" \
    -errts "$outdir/errts.${SUBJ}" \
    -cbucket "$outdir/all_betas.${SUBJ}" \
    -bucket "$outdir/stats.${SUBJ}"

  if [[ $? -ne 0 ]]; then
    echo "** 3dDeconvolve failed for $SUBJ, skipping"
    continue
  fi

  # Patch REML command to allow overwriting existing outputs
  sed -i '' 's/3dREMLfit/3dREMLfit -overwrite/' stats.REML_cmd

  tcsh -x stats.REML_cmd

  if [[ $? -ne 0 ]]; then
    echo "** 3dREMLfit failed for $SUBJ"
    continue
  fi

  echo "=== Done: $SUBJ ==="
done
