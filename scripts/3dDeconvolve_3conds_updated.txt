## Used this to create emotional-neutral constrast that was not included in afniprocpy

#!/usr/bin/env bash
ROOT="/Volumes/ELEMENTS/imagery/fmri"
TIMEDIR="$ROOT/time_cols"

for subdir in "$ROOT"/par_*; do
  SUBJ="${subdir##*/}"
  outdir="$subdir/PROC_cspline"

  [[ -d "$outdir" ]] || { echo "No PROC dir for $SUBJ, skipping"; continue; }
  cd "$outdir" || continue

  echo "=== Processing $SUBJ ==="

  # Detect O1 vs O2 prefix for this subject
  tdir="$TIMEDIR/$SUBJ"
  prefix=""
  for p in O1 O2; do
    if [[ -f "$tdir/${p}.pleasant.LOCAL.txt" ]]; then
      prefix="$p"
      break
    fi
  done

  if [[ -z "$prefix" ]]; then
    echo "** No timing files found for $SUBJ in $tdir, skipping"
    continue
  fi

  echo "   Using timing prefix: $prefix"

  # Dynamically build -ortvec args based on which mot_demean files exist
  ortvec_args=()
  for r in 01 02; do
    f="mot_demean.r${r}.1D"
    if [[ -f "$f" ]]; then
      ortvec_args+=( -ortvec "$f" "mot_demean_r${r}" )
    fi
  done

  if [[ ${#ortvec_args[@]} -eq 0 ]]; then
    echo "** No mot_demean files found for $SUBJ, skipping"
    continue
  fi

  3dDeconvolve \
    -input pb04.${SUBJ}.r*.scale+tlrc.HEAD \
    -censor censor_${SUBJ}_combined_2.1D \
    "${ortvec_args[@]}" \
    -polort 7 \
    -num_stimts 3 \
    -stim_times 1 "$tdir/${prefix}.pleasant.LOCAL.txt"   'CSPLINzero(0,31,15)' \
    -stim_label 1 pleasant \
    -stim_times 2 "$tdir/${prefix}.neutral.LOCAL.txt"    'CSPLINzero(0,31,15)' \
    -stim_label 2 neutral \
    -stim_times 3 "$tdir/${prefix}.unpleasant.LOCAL.txt" 'CSPLINzero(0,31,15)' \
    -stim_label 3 unpleasant \
    -iresp 1 iresp_pleasant.${SUBJ} \
    -iresp 2 iresp_neutral.${SUBJ} \
    -iresp 3 iresp_unpleasant.${SUBJ} \
    -fout -tout \
    -gltsym 'SYM: +pleasant -neutral'               -glt_label 1 pleasant-neutral \
    -gltsym 'SYM: +unpleasant -neutral'             -glt_label 2 unpleasant-neutral \
    -gltsym 'SYM: +pleasant -unpleasant'            -glt_label 3 pleasant-unpleasant \
    -gltsym 'SYM: +pleasant +unpleasant -2*neutral' -glt_label 4 emotional-neutral \
    -x1D X.xmat.1D \
    -xjpeg X.jpg \
    -x1D_uncensored X.nocensor.xmat.1D \
    -errts errts.${SUBJ} \
    -cbucket all_betas.${SUBJ} \
    -bucket stats.${SUBJ} \
    -x1D_stop

  if [[ $? -ne 0 ]]; then
    echo "** 3dDeconvolve failed for $SUBJ, skipping"
    continue
  fi

  # Patch REML command to allow overwriting existing outputs
  sed -i '' 's/3dREMLfit/3dREMLfit -overwrite/' stats.REML_cmd

  tcsh -x stats.REML_cmd

  if [[ $? -ne 0 ]]; then
    echo "** 3dREMLfit failed for $SUBJ"
    continue
  fi

  echo "=== Done: $SUBJ ==="
done
