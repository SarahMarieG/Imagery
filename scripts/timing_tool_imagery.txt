ROOT="/Users/sgardy/Documents/SarahData/Imagery/Data/time_cols"
CONDS=(pleasant neutral unpleasant)

cd "$ROOT" 

for subdir in "$ROOT"/par_*; do
  SUBJ="${subdir##*/}"
  echo "Processing subject: $SUBJ"
  timedir="$subdir"

if [[ -d "$timedir" ]]; then
    for c in "${CONDS[@]}"; do
    # Enable nullglob so a no-match doesn't yield a literal pattern
    shopt -s nullglob
    files=( "$timedir"/O*"$c".txt )
    shopt -u nullglob
    
   for timing in "${files[@]}"; do
      [ -f "$timing" ] || continue
      base="${timing%.txt}"
      output="${base}.LOCAL.txt"
 
timing_tool.py \
  -timing "$timing" \
  -global_to_local run_len \
  -run_len 1017 1017 \
  -write_timing "$output"

done
done
else
    echo "  (no timing dir) skipping"
  fi

  echo
done
