#creates mvm table and runs mvm for block 3dPM participant output (parametric modulation)

#!/usr/bin/env bash
ROOT="/Volumes/ELEMENTS/imagery"
OUTDIR="$ROOT/fmri/group_analysis/MVM_parametric_block"
TABLEFILE="$OUTDIR/mvm_parametric_table.txt"
mkdir -p "$OUTDIR"

SUBJS=($(ls -d "$ROOT"/participants/par_* 2>/dev/null | xargs -n1 basename))

# Build table with two within-subject factors: Emotion and Task
echo "Subj Emotion Task InputFile" > "$TABLEFILE"

for s in "${SUBJS[@]}"; do

  STATS_FILE="$ROOT/participants/$s/fmri/paramod_block/stats.${s}+tlrc"

  if [[ ! -f "${STATS_FILE}.HEAD" ]]; then
    echo "WARNING: missing $s file"
    continue
  fi

  # Extract the slope coefficients (#1_Coef) tracking alpha modulator
  echo "$s pleasant   read    \"${STATS_FILE}[pleasant_read#1_Coef]\""    >> "$TABLEFILE"
  echo "$s neutral    read    \"${STATS_FILE}[neutral_read#1_Coef]\""     >> "$TABLEFILE"
  echo "$s unpleasant read    \"${STATS_FILE}[unpleasant_read#1_Coef]\""  >> "$TABLEFILE"
  echo "$s pleasant   imagine \"${STATS_FILE}[pleasant_imagine#1_Coef]\"" >> "$TABLEFILE"
  echo "$s neutral    imagine \"${STATS_FILE}[neutral_imagine#1_Coef]\""  >> "$TABLEFILE"
  echo "$s unpleasant imagine \"${STATS_FILE}[unpleasant_imagine#1_Coef]\"">> "$TABLEFILE"
done

3dMVM \
  -prefix "$OUTDIR/results" \
  -jobs 8 \
  -mask "$ROOT/participants/par_01/fmri/PROC_cspline/full_mask.par_01+tlrc.HEAD" \
  -bsVars 1 \
  -wsVars "Emotion*Task" \
  -num_glt 3 \
  -gltLabel 1 Emotion_vs_Neutral \
    -gltCode 1 "Emotion : 0.5*pleasant +0.5*unpleasant -1*neutral" \
  -gltLabel 2 Read_vs_Imagine \
    -gltCode 2 "Task : 1*read -1*imagine" \
  -gltLabel 3 Emotion_X_Task_Interaction \
    -gltCode 3 "Emotion : 0.5*pleasant +0.5*unpleasant -1*neutral Task : 1*read -1*imagine" \
  -dataTable @"$TABLEFILE"
